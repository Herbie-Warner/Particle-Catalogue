191void Particle::set_four_momentum(double E, double px, double py, double pz)
192{
193 //Augments energies to adhere to known rest mass.
194 FourVector new_momentum(E, px, py, pz);
195 double four_momentum_sqr = new_momentum.dot_product(new_momentum);
196 double invariant_mass_squared = rest_mass * rest_mass;
197 double difference = std::abs(invariant_mass_squared - four_momentum_sqr);
198 static const double tolerance = 0.01;
199
200 if(difference / invariant_mass_squared > tolerance)
201 {
202 double E_squared = E*E;
203 double delta = four_momentum_sqr - invariant_mass_squared;
204 if(delta < E_squared)
205 {;
206 E = std::sqrt(E_squared - delta);
207 }
208 else
209 {
210 E = rest_mass;
211 px = 0;
212 py = 0;
213 pz = 0;
214 }
215 }
216 four_momentum->set_four_vector(E, px, py, pz);
217}
