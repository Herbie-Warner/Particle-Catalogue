57void Electron::set_calorimeter_deposit(double EM_1, double EM_2, double HAD_1, double HAD_2)
58{
59 double energy = get_E();
60 double sum = EM_1 + EM_2 + HAD_1 + HAD_2;
61 if(EM_1 < 0 || EM_2 < 0 || HAD_1 < 0 || HAD_2 < 0)
62 {
63 std::cout<<"CANNOT HAVE NEGATIVE ENERGIES: SETTING ALL EQUAL TO ELECTRON ENERGY/4"<<std::endl;
64 double energy_in_each_layer = energy / 4;
65 change_calorimeter(energy_in_each_layer, energy_in_each_layer, energy_in_each_layer, energy_in_each_layer);
66 }
67 else if(sum == energy)
68 {
69 change_calorimeter(EM_1, EM_2, HAD_1, HAD_2);
70 }
71 else if(sum < energy)
72 {
73 std::cout<<"AUGMENTING LAYER_1 ENERGY TO ACCOMADATE EXCESS ENERGY IN ELECTRON FOUR VECTOR"<<std::endl;
74 double difference = energy - sum;
75 EM_1 += difference;
76 change_calorimeter(EM_1, EM_2, HAD_1, HAD_2);
77 }
78 else
79 {
80 std::cout<<"ENERGY SET EXCEEDS ELECTRON ENERGY: SETTING ALL EQUAL TO ELECTRON ENERGY/4"<<std::endl;
81 double energy_in_each_layer = energy / 4;
82 change_calorimeter(energy_in_each_layer, energy_in_each_layer, energy_in_each_layer, energy_in_each_layer);
83 }
84}
