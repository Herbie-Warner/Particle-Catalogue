72 void distribute_four_momentum(const Particle& source, std::vector<std::unique_ptr<Particle>>& products)
73 {
74 if(products.size() != 2) {return;} //Method only works for 2 products as too many dof in three body decays
75
76 double rest_mass_0 = source.get_rest_mass();
77 double rest_mass_1 = products[0]->get_rest_mass();
78 double rest_mass_2 = products[1]->get_rest_mass();
79
80 if(rest_mass_0 <= rest_mass_1 + rest_mass_2) {return;} //Method only works if particles are non-virtual in source rest frame
81
82 double energy_1 = (std::pow(rest_mass_0, 2) + std::pow(rest_mass_1, 2) - std::pow(rest_mass_2, 2))/(2*rest_mass_0);
83 double momentum_1 = std::sqrt(std::pow(energy_1,2) - std::pow(rest_mass_1,2));
84
85 double energy_2 = std::pow(momentum_1, 2) + std::pow(rest_mass_2, 2);
86
87 products[0]->set_four_momentum(energy_1, momentum_1, 0, 0);
88 products[1]->set_four_momentum(energy_2, -momentum_1, 0, 0);
89
90 FourVector initial = source.get_four_vector();
91 double beta_x, beta_y, beta_z = 0;
92 initial.get_boost_to_lab_frame(beta_x, beta_y, beta_z);
93
94 products[0]->boost(-beta_x, -beta_y, -beta_z);
95 products[1]->boost(-beta_x, -beta_y, -beta_z);
96 }
