69void ParticleContainer::register_inheritance(ParticleIdentifier base, ParticleIdentifier derived)
70{
71 inheritance_map[base].insert(derived);
72 for(auto& [ancestor, descendants] : inheritance_map)
73 {
74 if(descendants.find(base) != descendants.end())
75 {
76 descendants.insert(derived);
77 }
78 }
79}
