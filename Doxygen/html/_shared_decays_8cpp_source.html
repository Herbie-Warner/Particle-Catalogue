1//SharedDecays.cpp
2//Defines the SharedDecays namespace. Initialises the shared decay ptrs by using templates
3//from the DecayTemplates class.
4//Branching ratios obtained from following sources:
5//Leptonic vs Hadronic for W: https://pdg.lbl.gov/2022/listings/particle_properties.html
6//CKM matrix: https://pdg.lbl.gov/2020/reviews/rpp2020-rev-ckm-matrix.pdf
7//ZBoson: https://www.hep.phy.cam.ac.uk/~thomson/lectures/partIIIparticles/Handout13_2009.pdf
8//Higgs: https://pdg.lbl.gov/2016/reviews/rpp2016-rev-higgs-boson.pdf
9//The reason the decays are not simply declared static in their relevant classes is so that
10//other particles may access and use them without having to decay through said particle. This permits
11//the hard coding of decays, required by the decays mandated, at the cost of prohibiting a nested
12//and physically accurate structure. By nested this refers to a tau going into a tau neutrino and a
13//W and then the W subsequently decaying and so on.
14//Herbie Warner 11/05/2024

87 void generate_w_boson_decays(int charge, std::shared_ptr<DecayChannelsVector>& decay)
88 {
89 //Leptonic decays
90 std::vector<LeptonType> leptons;
91 std::vector<QuarkFlavour> quarks;
92 if(charge == -1)
93 {
94 leptons = {LeptonType::Electron, LeptonType::Muon, LeptonType::Tau};
95 quarks = {QuarkFlavour::Anti_Up, QuarkFlavour::Anti_Charm, QuarkFlavour::Anti_Top};
96 }
97 else
98 {
99 leptons = {LeptonType::AntiElectron, LeptonType::AntiMuon, LeptonType::AntiTau};
100 quarks = {QuarkFlavour::Up, QuarkFlavour::Charm, QuarkFlavour::Top};
101 }

103 //Using template from DecayTemplates. This takes a vector of some members of some enum class, a function
104 //to generate another member of the same enum class from each variable in the vector, and a factory function
105 //to produce them as arguments. Also takes a function to calculate the branching ratio.
106 DecayTemplates::add_decay_channels_for_particle_pairs(leptons, LeptonUtilities::get_corresponding_anti_neutrino,
107 [](const auto& flavour) -> std::unique_ptr<Particle> {return factory_lepton(flavour);},
108 [](const auto&, const auto&) -> double {return w_branching_leptonic_all_flavours;},
109 decay);
110 //Branching ratio defined as above so the template is compatible with the cabibo branching which requires
111 //quark flavour arguments.
112
113 //Stop taus re decaying into taus. This can also be implemented to stop muons decaying into muons
114 //but muons are considered stable in this assignment. Similarly for top and so on. This would also be
115 //possible by at each vertex checking the invariant mass after does not exceed that before (for real
116 //particles that is).
117 auto& forbidden = decay->back().forbidden_source;
118 if(!forbidden)
119 {
120 forbidden = std::make_unique<std::unordered_set<ParticleUtilities::ParticleIdentifier>>();
121 }
122 if(charge == -1){forbidden->insert(ParticleUtilities::ParticleIdentifier::Tau);}
123 else {forbidden->insert(ParticleUtilities::ParticleIdentifier::AntiTau);}
124
125 //Hadronic decays
126 DecayTemplates::add_decay_channels_for_particle_pairs(quarks,
127 get_quark_w_pairs, [](auto flavour) -> std::unique_ptr<Particle> {return std::make_unique<Quark>(flavour);},
128 cabibo_branching, decay);
129 }

229 std::unique_ptr<Lepton> factory_lepton(const LeptonType& flavour)
230 {
231 //Factory function for creating lepton objects
232 if(LeptonUtilities::check_if_neutrino(flavour)) { return std::make_unique<Neutrino>(flavour); }
233 switch(flavour)
234 {
235 case LeptonType::Electron: return std::make_unique<Electron>(-1);
236 case LeptonType::AntiElectron: return std::make_unique<Electron>(1);
237 case LeptonType::Muon: return std::make_unique<Muon>(-1);
238 case LeptonType::AntiMuon: return std::make_unique<Muon>(1);
239 case LeptonType::Tau: return std::make_unique<Tau>(-1);
240 case LeptonType::AntiTau: return std::make_unique<Tau>(1);
241 default: break;
242 }
243 throw("UNSUPPORTED FLAVOUR FOR FACTORY LEPTON");
244 }

246 double cabibo_branching(QuarkUtilities::QuarkFlavour quark1, QuarkUtilities::QuarkFlavour quark2)
247 {
248 //It is less hard coded to instead compute row/column in the CKM matrix by checking each flavour
249 //in turn and assigning numbers this way, but provided the user does not change the order of flavours
250 //in the enum class this method suits and is much neater.
251 static double ckm_matrix[3][3] =
252 {
253 {0.97401, 0.22650, 0.00361},
254 {0.22636, 0.97320, 0.04053},
255 {0.00854, 0.03978, 0.999172}
256 };
257
258 int type_q1 = static_cast<int>(quark1);
259 int type_q2 = static_cast<int>(quark2);
260
261 if(type_q1 > type_q2)
262 {
263 int temp = type_q1;
264 type_q1 = type_q2;
265 type_q2 = temp;
266 }
267
268 type_q1 %= 3;
269 type_q2 %= 3;
270
271 double branching_ratio = ckm_matrix[type_q1][type_q2];
272 // /9 as 9 hadronic decays
273 return branching_ratio*branching_ratio/3 * w_branching_hadronic;
274 }

276 std::vector<QuarkFlavour> get_quark_w_pairs(QuarkFlavour base_quark)
277 {
278 //For template calls
279 if(base_quark == QuarkFlavour::Up || base_quark == QuarkFlavour::Charm || base_quark == QuarkFlavour::Top)
280 {
281 return {QuarkFlavour::Anti_Down, QuarkFlavour::Anti_Strange, QuarkFlavour::Anti_Bottom};
282 }
283 else if(base_quark == QuarkFlavour::Anti_Up || base_quark == QuarkFlavour::Anti_Charm || base_quark == QuarkFlavour::Anti_Top)
284 {
285 return {QuarkFlavour::Down, QuarkFlavour::Strange, QuarkFlavour::Bottom};
286 }
287 else
288 {
289 return {};
290 }
