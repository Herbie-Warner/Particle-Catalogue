50bool QuantumNumberVisitor::spin_validation(double new_spin) const
51{
52 //As the exact equality of spin of particles does not count for equality, checks if system goes from spin 1/2 -> spin 1
53 double current_spin = total_spin;
54 current_spin *= 2;
55 int int_current_spin = static_cast<int>(current_spin);
56 new_spin *= 2;
57 int int_new_spin = static_cast<int>(new_spin);
58
59 return (int_new_spin % 2) == (int_current_spin % 2);
60}
